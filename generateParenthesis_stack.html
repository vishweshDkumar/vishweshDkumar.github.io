def generateParenthesis(N):
     if N == 0: return ['']
     ans = [] # watchvar ans
     for c in range(N):
         for left in generateParenthesis(c):
             for right in generateParenthesis(N-1-c):
                 ans.append('({}){}'.format(left, right))
