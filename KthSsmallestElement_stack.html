class TreeNode:

         self.val = val
         self.left = left
         self.right = right
     def __str__(self):
         s = str(self.val)
         if self.left:
             s = str(self.left) + " "+s
         if self.right:
             s += " " + str(self.right)
         return s
     def __repr__(self):
         return self.__str__()
     def __eq__(self,other):
